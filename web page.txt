  <script>
    document.getElementById("omrForm").addEventListener("submit", function(event) {
      event.preventDefault();

      const resultDiv = document.getElementById("result");
      resultDiv.style.display = "block";

      // Get student info fields
      const name = document.getElementById("name").value.trim();
      const studentClass = document.getElementById("class").value.trim();
      const subject = document.getElementById("subject").value.trim();
      const testDate = document.getElementById("date").value;

      // Collect answers for all 15 questions
      const answers = {};
      for(let i = 1; i <= 15; i++) {
        const selected = document.querySelector(`input[name="q${i}"]:checked`);
        answers[`প্রশ্ন ${i}`] = selected ? selected.value : "উত্তর দেওয়া হয়নি";
      }

      // Create display HTML
      let output = `<h3>ছাত্র/ছাত্রী তথ্য</h3>
      <p>নাম: ${name}</p>
      <p>শ্রেণি: ${studentClass}</p>
      <p>বিষয়: ${subject}</p>
      <p>পরীক্ষার তারিখ: ${testDate}</p>
      <hr/>
      <h3>উত্তরসমূহ:</h3>
      <ul>`;

      for (const [q, a] of Object.entries(answers)) {
        output += `<li>${q}: ${a}</li>`;
      }
      output += "</ul>";

      resultDiv.innerHTML = output;
    });
  </script>
